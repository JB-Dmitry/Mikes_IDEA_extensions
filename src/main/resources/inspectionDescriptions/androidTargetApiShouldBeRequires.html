Reports use of Android @TargetApi annotation.

@TargetApi just suppresses minSdk errors inside annotated declaration, while @RequiresApi also reports minSdk errors for usages of annotated declaration. 


@TargetApi(99) fun target() {}
@RequiresApi(99) fun requires() {}

fun whatever() {
    target() // unfortunately, not an error
    requires() // red underline, as desired
}
